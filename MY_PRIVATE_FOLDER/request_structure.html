CREATE (CREATE)

Request Structure (to Server) Comment
("/api/add/department") 
{
    name: (type: String),
    budget: (type: Integer),
    startDate: (type: Date),
    admistrator: (type: String)
}
("/api/add/course") 
{
    title: (type: String),
    credits: (type: String),
    departmentId: (type: Integer)
}
("/api/add/instructor") 
{
    lastName: (type: String),
    firstName: (type: String),
    type: 'instructor' (type: String),
    hireDate: (type: Date),
    enrollmentDate: (type: Date)
}
("/api/add/student") 
{
    lastName: (type: String),
    firstName: (type: String),
    type: 'student' (type: String),
    hireDate: (type: Date),
    enrollmentDate: (type: Date)
}
("/api/add/course-instructor") 
{
    instructorId: (type: String, FK),
    courseId: (type: String, FK)
}
("/api/add/course-student") 
{
        studentId: (type: String, FK),
        courseId: (type: String, FK),
        grade: (type: String)
    }
("/api/set/course/online") 
{
    courseId: (type: Integer, FK),
    url: (type: String)
}
("/api/set/course/onsite") 
{
    courseId: (type: Integer, FK),
    location: (type: String),
    day: (type: String),
    time: (type: String)
}
